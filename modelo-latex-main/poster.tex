% Contexto
\posterbox[adjusted title = Context, colback=white]
          {name=context, below=titlebox, column=1, span=6, rowspan=0.8}{
    Embedding models encode semantic meaning into numerical vectors, powering modern NLP systems. They are widely used in software engineering for tasks like bug detection and requirements traceability.
}

% Problema
\posterbox[adjusted title = Problem, colback=white]
          {name=problem, below=titlebox, column=7, span=6, rowspan=0.8}{
    \textbf{Research Question:} Does the choice of embedding model influence task outcomes?
    
    \vspace{0.2em}
    New models are released rapidly with improvement claims, but lack studies evaluating real-world performance differences.
}

% Metodologia - Dataset
\posterbox[adjusted title = Data and Models, colback=white]
          {name=dataset, below=context, column=1, span=6, rowspan=0.8}{
    \small
    \textbf{Data:} DeeperMatcher \autocite{pilone2024deepermatcher}
    
    \begin{itemize}
        \item 196 projects, 87K issues, 1.3M reviews
        \item 4 projects selected
        \item Human-validated scores (1--5)
    \end{itemize}
    
    \vspace{0.2em}
    \textbf{Models:} Jina 3, OpenAI Large, OpenAI Small, Gemini Text Embedding 004, Cohere Embed v4.0
}

% Metodologia - Processo
\posterbox[adjusted title = Method, colback=white]
          {name=method, below=context, column=7, span=6, rowspan=0.8}{
    \small
    \raggedright
    \textit{Extending DeeperMatcher \autocite{pilone2024deepermatcher}:}
    
    \vspace{0.2em}
    \begin{enumerate}
        \item Embed issues \& reviews
        \item Compute cosine similarity
        \item Rank \& select top 10
        \item Compare with human scores
    \end{enumerate}
}
